var express = require("express");
var connection = require("../database");
var router = express.Router();

router.delete("/", function(req, res, next) {
  switch (req.query.type) {
    case "table":
      //Delete everything in the table given within the request
      var sql = `DELETE FROM ${req.query.table}`;
      connection.query(sql, (error, result) => {
        if (error) {
          res.sendStatus(500);
          return console.error("Error: " + error.message);
        } else {
          res.sendStatus(200);
        }
      });
      break;
    case "sensor":
      //Delete a sensor that matches the ID
      var sql = `DELETE FROM SensorOverview WHERE ID='${req.query.id}'`;
      connection.query(sql, (error, result) => {
        if (error) {
          res.sendStatus(500);
          return console.error("Error: " + error.message);
        } else {
          res.sendStatus(200);
        }
      });
      break;
    case "data":
      // Delete a specific value from the sensor data
      var sql = `DELETE FROM SensorData WHERE ID='${req.query.id}' AND Value = '${req.query.val}' AND Timestamp = '${req.query.time}'`;
      console.log(req.query.time);
      connection.query(sql, (error, result) => {
        if (error) {
          res.sendStatus(500);
          return console.error("Error: " + error.message);
        } else {
          res.sendStatus(200);
          console.log(result);
        }
      });
      break;

    default:
      break;
  }
});

module.exports = router;
